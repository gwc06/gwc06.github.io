* bash脚本
```
#!/bin/bash
# 重置filenames.txt和dirnames.txt
> filenames.txt
> dirnames.txt
# 将ls的结果保存到变量name_hw中
name_hw=`ls`
# 遍历name_hw
for val in $name_hw
do
# 若val是文件，则将该文件名追加写入文件filenames.txt
if [ -f $val ];then
echo "f$val" >> filenames.txt
# 若val是文件夹，则将该子文件夹名追加写入文件dirnames.txt
elif [ -d $val ]; then
echo "$val" >> dirnames.txt
fi
done
exit 0
```
