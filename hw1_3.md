* bash脚本
```
#!/bin/bash
# 重置filenames.txt和dirnames.txt
> filenames.txt
> dirnames.txt
# 将ls的结果保存到变量name_hw中
name_hw=$(ls bash_homework)
# 在bash_homework下遍历name_hw进行判断
for val in $name_hw
do
cd bash_homework
# 若val是文件，则将该文件名追加写入上级文件夹的文件filenames.txt中
if [ -f $val ];then
cd ..
echo "$val" >> filenames.txt
# 若val是文件夹，则将该子文件夹名追加写入上级文件夹的文件dirnames.txt中
elif [ -d $val ]; then
cd ..
echo "$val" >> dirnames.txt
fi
done

exit 0
```  
